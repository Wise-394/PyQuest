class Hero:
     def __init__(self, name, health):
         self.name = name
         self.health = health
     
     def is_alive(self):
         return self.health > 0
     
     def heal(self, amount):
         self.health += amount
     
     def take_damage(self, amount):
         self.health -= amount
         if self.health < 0:
             self.health = 0
